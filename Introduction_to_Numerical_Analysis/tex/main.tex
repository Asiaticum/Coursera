    \mathbf{\Lambda_{1}^{-1}\Lambda_{2}^{-1}\cdots\Lambda_{{\it m}-1}^{-1}}=
    \begin{pmatrix}
    1  & 0 & 0 & \cdots & 0 \\
    \gamma_{21}  & 1 & 0 & \cdots & 0 \\

\begin{eqnarray*}

    \vdots
\end{eqnarray*}

\begin{equation*}
